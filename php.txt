<?php
use PHPMailer\PHPMailer\PHPMailer;
use PHPMailer\PHPMailer\Exception;

require __DIR__ . '/../vendor/autoload.php';

/**
 * Sends an email with optional HTML body and attachment.
 *
 * @param string $to Recipient email address
 * @param string $subject Email subject
 * @param string $body Email body (HTML or plain text)
 * @param string|null $attachmentPath File path to attach (optional)
 * @param bool $isHtml Whether the body is HTML (default: false)
 * @return bool True if sent successfully, false otherwise
 */
function sendEmail($to, $subject, $body, $attachmentPath = null, $isHtml = false) {
    $mail = new PHPMailer(true);

    try {
        // === Gmail SMTP Settings ===
        $mail->isSMTP();
        $mail->Host       = 'smtp.gmail.com';
        $mail->SMTPAuth   = true;
        $mail->Username   = 'shineakakpo08@gmail.com';      // ðŸ”¹ Replace with your Gmail
        $mail->Password   = 'uplu fskx kwae mhin'; // ðŸ”¹ Replace with App Password
        $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;
        $mail->Port       = 587;

        // === Sender & Recipient ===
        $mail->setFrom('your-email@gmail.com', 'GEBSCO School'); // Sender
        $mail->addAddress($to); // Recipient

        // === Content ===
        $mail->isHTML($isHtml);
        $mail->Subject = $subject;
        $mail->Body    = $body;

        if (!$isHtml) {
            $mail->AltBody = strip_tags($body);
        }

        // === Attachment (optional) ===
        if ($attachmentPath && file_exists($attachmentPath)) {
            $mail->addAttachment($attachmentPath);
        }

        return $mail->send();
    } catch (Exception $e) {
        error_log("Email sending failed: {$mail->ErrorInfo}");
        return false;
    }
}
